#!/usr/bin/env bash
set -e

cpu_usage=0
load_1=0
load_5=0
load_15=0
ram_used=0
ram_free=0
uptime_sec=0
disk_json="[]"
net_json="[]"
gpu_json="[]"

if [ "$(uname -s)" = "Darwin" ]; then
	cpu_usage="$(top -l 1 -n 0 | awk '/CPU usage/ { gsub(/%/,""); print $3 + $5 }')"

	loads=$(uptime | awk -F'load averages?: ' '{print $2}')
	load_1=$(echo "$loads" | awk '{print $1}')
	load_5=$(echo "$loads" | awk '{print $2}')
	load_15=$(echo "$loads" | awk '{print $3}')

	page_size=$(sysctl -n hw.pagesize)
	stats=$(vm_stat)
	free_pages=$(echo "$stats" | awk '/Pages free/ {gsub(/\./,"",$3); print $3}')
	inactive_pages=$(echo "$stats" | awk '/Pages inactive/ {gsub(/\./,"",$3); print $3}')
	speculative_pages=$(echo "$stats" | awk '/Pages speculative/ {gsub(/\./,"",$3); print $3}')
	purgeable_pages=$(echo "$stats" | awk '/Pages purgeable/ {gsub(/\./,"",$3); print $3}')
	ram_total=$(sysctl -n hw.memsize)
	ram_free=$(( (free_pages + inactive_pages + speculative_pages + purgeable_pages) * page_size ))
	ram_used=$(( ram_total - ram_free ))

	boot=$(sysctl -n kern.boottime | awk '{ gsub(/[{},]/,""); print $3 }')
	uptime_sec=$(( $(date +%s) - boot ))

	disk_json="$(df -kP | awk 'NR>1 && $1 ~ /^\/dev\// { dev=$1; sub(/s[0-9]+.*$/,"",dev); used[dev]=$3*1024; free[dev]=$4*1024 } END { first=""; printf "["; for (d in used) { if (first) printf ","; printf "{\"device\":\"%s\",\"used\":%s,\"free\":%s}", d, used[d], free[d]; first=1 } printf "]" }')"

	rx1=$(netstat -ibn | awk 'NR>1 && $1 ~ /^en[0-9]+$/ && !seen[$1]++ { printf "%s:%s:%s\n", $1, $7, $10 }')
	sleep 1
	rx2=$(netstat -ibn | awk 'NR>1 && $1 ~ /^en[0-9]+$/ && !seen[$1]++ { printf "%s:%s:%s\n", $1, $7, $10 }')

	net_json="$(paste -d '|' <(echo "$rx1") <(echo "$rx2") | awk -F'|' '
		{
			split($1, a, ":"); split($2, b, ":");
			iface=a[1]; rx1=a[2]; tx1=a[3]; rx2=b[2]; tx2=b[3];
			if((rx2>0 || tx2>0)) {
				if(first) printf ","; else printf "[";
				printf "{\"interface\":\"%s\",\"rx_bytes\":%s,\"tx_bytes\":%s,\"rx_per_sec\":%s,\"tx_per_sec\":%s}", iface, rx2, tx2, (rx2-rx1), (tx2-tx1);
				first=1;
			}
		}
		END { if(first) printf "]"; else printf "[]" }')"

elif [ "$(uname -s)" = "Linux" ]; then
	cpu1=$(awk '/^cpu / { total=0; for(i=2;i<=NF;i++) total+=$i; print total, $5 }' /proc/stat)
	net1=$(cat /proc/net/dev)
	sleep 1
	cpu2=$(awk '/^cpu / { total=0; for(i=2;i<=NF;i++) total+=$i; print total, $5 }' /proc/stat)
	net2=$(cat /proc/net/dev)

	cpu_usage=$(echo "$cpu1 $cpu2" | awk '{ dt=$3-$1; di=$4-$2; if(dt>0) printf "%.2f", (1-di/dt)*100; else print "0.00" }')

	load_1=$(awk '{print $1}' /proc/loadavg)
	load_5=$(awk '{print $2}' /proc/loadavg)
	load_15=$(awk '{print $3}' /proc/loadavg)

	ram_total=$(awk '/MemTotal/ { print $2 * 1024 }' /proc/meminfo)
	ram_free=$(awk '/MemAvailable/ { print $2 * 1024 }' /proc/meminfo)
	ram_used=$(( ram_total - ram_free ))

	uptime_sec=$(awk '{ print int($1) }' /proc/uptime)

	disk_json="$(df -kP | awk 'NR>1 && $1 ~ /^\/dev\// { if(first) printf ","; else printf "["; printf "{\"device\":\"%s\",\"used\":%s,\"free\":%s}", $1, $3*1024, $4*1024; first=1 } END { if(first) printf "]"; else printf "[]" }')"

	net_json="$(paste <(echo "$net1") <(echo "$net2") | awk '
		NR>2 {
			gsub(/:/,"");
			iface=$1; rx1=$2; tx1=$10; rx2=$(NF-15); tx2=$(NF-7);
			if(iface!="lo" && (rx2>0 || tx2>0)) {
				if(first) printf ","; else printf "[";
				printf "{\"interface\":\"%s\",\"rx_bytes\":%s,\"tx_bytes\":%s,\"rx_per_sec\":%s,\"tx_per_sec\":%s}", iface, rx2, tx2, (rx2-rx1), (tx2-tx1);
				first=1;
			}
		}
		END { if(first) printf "]"; else printf "[]" }')"

	if command -v nvidia-smi >/dev/null 2>&1; then
		gpu_json="$(nvidia-smi --query-gpu=utilization.gpu,memory.used,temperature.gpu --format=csv,noheader,nounits | awk -F', ' '{ if(first) printf ","; else printf "["; printf "{\"usage\":%d,\"vram_used\":%d,\"temp\":%d}", $1, $2, $3; first=1 } END { if(first) printf "]"; else printf "[]" }')"
	fi
fi

cat <<JSON
{
	"cpu": {
		"usage": $cpu_usage,
		"load_1": $load_1,
		"load_5": $load_5,
		"load_15": $load_15
	},
	"ram": {
		"used": $ram_used,
		"free": $ram_free
	},
	"uptime": $uptime_sec,
	"disk": $disk_json,
	"network": $net_json,
	"gpu": $gpu_json
}
JSON
